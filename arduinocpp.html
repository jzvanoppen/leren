Arduino Oefeningen

Oefening 1: LED laten knipperen
▶

Laat een LED knipperen met een interval van 1 seconde.


// LED op pin 13 knipperen
void setup() {
  pinMode(13, OUTPUT);
}

void loop() {
  digitalWrite(13, HIGH);
  delay(1000);
  digitalWrite(13, LOW);
  delay(1000);
}
Oefening 2: Drukknop LED aan/uit
▶

Gebruik een drukknop om een LED te schakelen.


int button = 2;
int led = 8;

void setup() {
  pinMode(button, INPUT_PULLUP);
  pinMode(led, OUTPUT);
}

void loop() {
  if (digitalRead(button) == LOW) {
    digitalWrite(led, HIGH);
  } else {
    digitalWrite(led, LOW);
  }
}
Oefening 3: Fade een LED met PWM
▶

Laat een LED geleidelijk faden op pin 9.


int led = 9;

void setup() {
  pinMode(led, OUTPUT);
}

void loop() {
  for (int i = 0; i <= 255; i++) {
    analogWrite(led, i);
    delay(5);
  }
  for (int i = 255; i >= 0; i--) {
    analogWrite(led, i);
    delay(5);
  }
}
Oefening 4: Temperatuur meten (LM35)
▶

Lees een LM35-sensor uit en print de temperatuur in °C.


int sensor = A0;

void setup() {
  Serial.begin(9600);
}

void loop() {
  int value = analogRead(sensor);
  float voltage = value * (5.0 / 1023.0);
  float temp = voltage * 100; // LM35: 10mV per °C

  Serial.print("Temperatuur: ");
  Serial.print(temp);
  Serial.println(" °C");

  delay(500);
}
